# 卡牌勘误平台 实现计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 构建一个卡牌勘误管理 Web 平台，以本地 .card 数据库为索引基准，关联 TTS 中英文卡图数据，提供三栏对比编辑、批量审核、精灵图生成与发布流程。

**Architecture:** FastAPI 单进程后端（同步 + 线程池处理图像计算），React TypeScript SPA 前端（Ant Design + Monaco Editor），PostgreSQL 数据库，arkham-card-maker 作为外部 pip 依赖。TranslationBag 的精灵图生成和上传逻辑内聚至项目中。

**Tech Stack:** Python 3.12+ / FastAPI / SQLAlchemy / Alembic / React 18 / TypeScript / Vite / Ant Design 5 / Monaco Editor / PostgreSQL / Pillow

```typescript
  const token = localStorage.getItem('token');
  if (token) {
    config.headers.Authorization = `Bearer ${token}`;
  }
  return config;
});

client.interceptors.response.use(
  (response) => response,
  (error) => {
    if (error.response?.status === 401) {
      localStorage.removeItem('token');
      window.location.href = '/login';

    return Promise.reject(error);
  }
);
```

```python
from datetime import datetime, timedelta
from jose import jwt, JWTError
from passlib.context import CryptContext
from app.config import settings

pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")

def hash_password(password: str) -> str:
    return pwd_context.hash(password)

def verify_password(plain: str, hashed: str) -> bool:
    return pwd_context.verify(plain, hashed)

def create_access_token(user_id: int, role: str) -> str:
    expire = datetime.utcnow() + timedelta(minutes=settings.access_token_expire_minutes)
    payload = {"sub": str(user_id), "role": role, "exp": expire}
    return jwt.encode(payload, settings.secret_key, algorithm="HS256")

def decode_token(token: str) -> dict | None:
    try:
        payload = jwt.decode(token, settings.secret_key, algorithms=["HS256"])
        return {"user_id": int(payload["sub"]), "role": payload["role"]}
    except JWTError:
        return None
```

```python
import json
import hashlib
import os
from pathlib import Path
from dataclasses import dataclass

@dataclass
class ScannedCard:
    arkhamdb_id: str
    face: str               # "a", "b", "a-c"
    relative_path: str
    category: str           # "剧本卡", "玩家卡", "重返卡"
    cycle: str              # 如 "01_基础游戏"
    name_zh: str
    card_type: str
    is_double_sided: bool
    content_hash: str
    last_modified: str
    content_json: dict      # 不含 picture_base64

def scan_card_database(root: Path) -> list[ScannedCard]:
    """
    遍历卡牌数据库目录，返回所有 .card 文件的扫描结果。
    目录结构: 剧本卡/01_基础游戏/01109_a.card
    """
    cards: list[ScannedCard] = []
    for card_file in sorted(root.rglob("*.card")):
        if card_file.name.startswith("."):
            continue
        relative = card_file.relative_to(root)
        parts = relative.parts  # 如 ("剧本卡", "01_基础游戏", "01109_a.card")

        if len(parts) < 3:
            continue

        category = parts[0]
        cycle = parts[1]
        filename = parts[2]

        # 解析文件名: XXXXX_Y.card → arkhamdb_id + face
        stem = filename.replace(".card", "")
        if "_" not in stem:
            continue
        arkhamdb_id, face = stem.rsplit("_", 1)
        if face not in ("a", "b", "a-c"):
            continue

        # 计算文件哈希
        content_bytes = card_file.read_bytes()
        content_hash = hashlib.sha256(content_bytes).hexdigest()
        last_modified = str(os.path.getmtime(str(card_file)))

        # 解析 JSON
        try:
            data = json.loads(content_bytes)
        except json.JSONDecodeError:
            continue

        # 剥离 picture_base64
        clean_data = {k: v for k, v in data.items() if k != "picture_base64"}

        card = ScannedCard(
            arkhamdb_id=arkhamdb_id,
            face=face,
            relative_path=str(relative),
            category=category,
            cycle=cycle,
            name_zh=data.get("name", ""),
            card_type=data.get("type", ""),
            is_double_sided=data.get("double_sided", False) or "back" in data,
            content_hash=content_hash,
            last_modified=last_modified,
            content_json=clean_data,
        )
        cards.append(card)

    return cards

def detect_double_sided(cards: list[ScannedCard]) -> set[str]:
    """检测哪些 arkhamdb_id 有双面（同时存在 a 和 b）"""
    face_map: dict[str, set[str]] = {}
    for c in cards:
        face_map.setdefault(c.arkhamdb_id, set()).add(c.face)
    return {aid for aid, faces in face_map.items() if faces & {"a", "b"}}

def load_card_content(root: Path, relative_path: str) -> dict | None:
    """加载单张 .card 的完整内容（不含 picture_base64）"""
    filepath = root / relative_path
    if not filepath.exists():
        return None
    data = json.loads(filepath.read_bytes())
    return {k: v for k, v in data.items() if k != "picture_base64"}
```

```python
import json
from pathlib import Path
from dataclasses import dataclass
from typing import Optional

@dataclass
class ParsedTTSCard:
    arkhamdb_id: str          # 从 GMNotes 提取
    source: str               # "英文" | "中文"
    relative_json_path: str
    card_id: int
    deck_key: str
    face_url: str
    back_url: str
    grid_width: int
    grid_height: int
    grid_position: int
    unique_back: bool
    nickname: str
    guid: str
    sideways_card: bool

def parse_gmnotes(gmnotes_raw: str) -> dict:
    """解析 GMNotes 字段（JSON 字符串 → dict）"""
    try:
        return json.loads(gmnotes_raw)
    except json.JSONDecodeError:
        return {}

def extract_arkhamdb_id(gmnotes_raw: str) -> Optional[str]:
    """从 GMNotes 提取 arkhamdb id"""
    data = parse_gmnotes(gmnotes_raw)
    return data.get("id")

def parse_tts_card_json(filepath: Path, source: str, root: Path) -> Optional[ParsedTTSCard]:
    """解析单个 TTS Card JSON 文件"""
    if not filepath.suffix == ".json":
        return None
    try:
        data = json.loads(filepath.read_text(encoding="utf-8"))
    except (json.JSONDecodeError, UnicodeDecodeError):
        return None

    # 只处理 Card 对象（跳过 Bag、Deck 等）
    if data.get("Name") != "Card":
        return None

    gmnotes = data.get("GMNotes", "{}")
    arkhamdb_id = extract_arkhamdb_id(gmnotes)
    if not arkhamdb_id:
        return None

    card_id = data.get("CardID", 0)
    card_id_str = str(card_id)

    # 确定 deck_key (CardID 去除末两位的前缀)
    if len(card_id_str) <= 2:
        return None
    deck_key = card_id_str[:-2]
    grid_position = int(card_id_str[-2:])

    custom_deck = data.get("CustomDeck", {})
    sheet = custom_deck.get(deck_key, {})

    return ParsedTTSCard(
        arkhamdb_id=arkhamdb_id,
        source=source,
        relative_json_path=str(filepath.relative_to(root)),
        card_id=card_id,
        deck_key=deck_key,
        face_url=sheet.get("FaceURL", ""),
        back_url=sheet.get("BackURL", ""),
        grid_width=sheet.get("NumWidth", 10),
        grid_height=sheet.get("NumHeight", 1),
        grid_position=grid_position,
        unique_back=sheet.get("UniqueBack", False),
        nickname=data.get("Nickname", ""),
        guid=data.get("GUID", ""),
        sideways_card=data.get("SidewaysCard", False),
    )

def scan_tts_directory(root: Path, source: str) -> list[ParsedTTSCard]:
    """递归遍历 TTS 卡图目录，解析所有 Card JSON"""
    cards: list[ParsedTTSCard] = []
    for json_file in sorted(root.rglob("*.json")):
        result = parse_tts_card_json(json_file, source, root)
        if result:
            cards.append(result)
    return cards

def find_shared_backs(cards: list[ParsedTTSCard], source: str) -> list[dict]:
    """提取共享卡背（unique_back=false 的卡背 URL）"""
    back_map: dict[str, dict] = {}  # keyed by (deck_key, back_url)
    for c in cards:
        if not c.unique_back and c.back_url:
            key = c.back_url
            if key not in back_map:
                back_map[key] = {
                    "back_url": c.back_url,
                    "source": source,
                    "deck_key": c.deck_key,
                }
    return list(back_map.values())
```

```python
import httpx
from pathlib import Path
from PIL import Image
from io import BytesIO

CARD_W = 750
CARD_H = 1050

def calc_grid_coords(grid_position: int, grid_width: int) -> tuple[int, int, int, int]:
    """计算卡牌在精灵图中的像素坐标范围"""
    row = grid_position // grid_width
    col = grid_position % grid_width
    x = col * CARD_W
    y = row * CARD_H
    return (x, y, x + CARD_W, y + CARD_H)

def download_and_cut_sheet(
    sheet_url: str,
    grid_position: int,
    grid_width: int,
    grid_height: int,
    cache_dir: Path,
    cache_key: str,
) -> str | None:
    """
    下载一张精灵图大图，裁切出指定位置的单张卡图，缓存到本地。
    返回缓存文件路径。
    """
    expected_path = cache_dir / f"{cache_key}.jpg"
    if expected_path.exists():
        return str(expected_path)

    try:
        resp = httpx.get(sheet_url, timeout=30, follow_redirects=True)
        resp.raise_for_status()
        sheet_img = Image.open(BytesIO(resp.content))
    except Exception:
        return None

    x1, y1, x2, y2 = calc_grid_coords(grid_position, grid_width)
    # 确保不超出图片边界
    x2 = min(x2, sheet_img.width)
    y2 = min(y2, sheet_img.height)

    card_img = sheet_img.crop((x1, y1, x2, y2))
    expected_path.parent.mkdir(parents=True, exist_ok=True)
    card_img.save(expected_path, "JPEG", quality=90)
    return str(expected_path)

def cache_all_tts_images(
    tts_cards: list,  # list[ParsedTTSCard] 或 list[TTSCardImage]
    cache_dir: Path,
) -> dict[str, tuple[str | None, str | None]]:
    """
    批量预切割所有 TTS 卡图。
    返回 {arkhamdb_id: (front_cache_path, back_cache_path)}
    """
    result = {}
    for card in tts_cards:
```

```python
import json
import tempfile
from pathlib import Path
from app.config import settings

def render_card_preview(card_content: dict, output_dir: Path, filename: str) -> str | None:
    """
    将 .card JSON 内容渲染为卡牌预览图。
    使用 arkham-card-maker 的 CardRenderer API。
    返回输出图片路径。
    """
    try:
        from arkham_card_maker import CardRenderer, RenderOptions

        renderer = CardRenderer()
        # 将内容写入临时 .card 文件
        temp_card = output_dir / f"{filename}.card"
        temp_card.write_text(json.dumps(card_content, ensure_ascii=False), encoding="utf-8")

        options = RenderOptions(
            dpi=150,         # 预览用 150 DPI 即可，节省时间
            format="JPEG",
            bleed=0,
            double_sided=("back" in card_content or card_content.get("double_sided", False)),
        )
        result = renderer.render(str(temp_card), options)

        if card_content.get("double_sided") or "back" in card_content:
            output_path = output_dir / f"{filename}_a.jpg"
            result.save_all(str(output_dir / filename))
        else:
            output_path = output_dir / f"{filename}.jpg"
            result.save(str(output_path))

        # 清理临时 .card 文件
        temp_card.unlink(missing_ok=True)
        return str(output_path)
    except ImportError:
        return None
    except Exception as e:
        # 记录错误但不崩溃
        print(f"渲染失败: {e}")
        return None
```

```python
import math
import os
from pathlib import Path
from PIL import Image

CARD_W = 750
CARD_H = 1050
MAX_COLS = 10  # 每行最多 10 张
MAX_BYTES = 10_485_670  # Steam 限制约 10MB

def create_decksheet(
    img_path_list: list[str],
    grid_size: tuple[int, int] | None = None,
    output_path: str = "",
    quality: int = 90,
) -> str:
    """
    将多张单卡图片拼合成精灵图大图。
    迁移自 TranslationBag main.py create_decksheet()。

    img_path_list: 单张卡图路径列表
    grid_size: (rows, cols)，为 None 则自动计算
    output_path: 输出文件路径
    quality: JPEG 质量 1-100
    """
    if not img_path_list:
        raise ValueError("图片列表为空")

    count = len(img_path_list)
    if grid_size is None:
        cols = min(count, MAX_COLS)
        rows = math.ceil(count / cols)
    else:
        rows, cols = grid_size

    canvas_w = cols * CARD_W
    canvas_h = rows * CARD_H
    canvas = Image.new("RGB", (canvas_w, canvas_h), (0, 0, 0))

    for idx, img_path in enumerate(img_path_list):
        row = idx // cols
        col = idx % cols
        x = col * CARD_W
        y = row * CARD_H

        try:
            img = Image.open(img_path)
            # 横卡旋转 270 度
            if img.width > img.height:
                img = img.rotate(270, expand=True)
            # 缩放到标准尺寸
            img = img.resize((CARD_W, CARD_H), Image.LANCZOS)
            canvas.paste(img, (x, y))
        except Exception:
            # 某张图片加载失败，跳过并留空
            continue

    output_path = str(output_path)
    os.makedirs(os.path.dirname(output_path), exist_ok=True)

    # 逐步降质直到文件大小在限制内
    current_quality = quality
    while current_quality >= 5:
        canvas.save(output_path, "JPEG", quality=current_quality)
        size = os.path.getsize(output_path)
        if size <= MAX_BYTES:
            break
        current_quality -= 5

    return output_path

def group_cards_by_sheet(
    card_images: list[dict],  # [{arkhamdb_id, front_path, back_path}]
    max_per_sheet: int = 30,
) -> list[dict]:
    """
    将卡牌按 sheet 分组。
    返回: [{sheet_name, front_images: [], back_images: [], arkhamdb_ids: []}]
    """
    sheets = []
    current_sheet = {
        "front_images": [],
        "back_images": [],
        "arkhamdb_ids": [],
    }

    for card in card_images:
        if len(current_sheet["arkhamdb_ids"]) >= max_per_sheet:
            sheets.append(current_sheet)
            current_sheet = {"front_images": [], "back_images": [], "arkhamdb_ids": []}

        current_sheet["front_images"].append(card.get("front_path", ""))
        current_sheet["back_images"].append(card.get("back_path", ""))
        current_sheet["arkhamdb_ids"].append(card["arkhamdb_id"])

    if current_sheet["arkhamdb_ids"]:
        sheets.append(current_sheet)

    # 为每个 sheet 命名
    for i, sheet in enumerate(sheets):
        start_id = sheet["arkhamdb_ids"][0]
        end_id = sheet["arkhamdb_ids"][-1]
        sheet["sheet_name"] = f"SheetZH{start_id}-{end_id}"

    return sheets
```

```python
import base64
import httpx
from pathlib import Path
from abc import ABC, abstractmethod

class ImageUploader(ABC):
    @abstractmethod
    async def check_exists(self, filename: str) -> str | None:
        """检查文件是否已上传，返回已有 URL"""
        pass

    @abstractmethod
    async def upload(self, filepath: str, filename: str) -> str | None:
        """上传文件，返回 URL"""
        pass

class CloudinaryUploader(ImageUploader):
    def __init__(self, cloud_name: str, api_key: str, api_secret: str):
        self.cloud_name = cloud_name
        self.api_key = api_key
        self.api_secret = api_secret

    async def check_exists(self, filename: str) -> str | None:
        import cloudinary
        cloudinary.config(
            cloud_name=self.cloud_name,
            api_key=self.api_key,
            api_secret=self.api_secret,
        )
        try:
            result = cloudinary.Search().expression(f"public_id:{filename}").execute()
            if result.get("resources"):
                return result["resources"][0]["secure_url"]
        except Exception:
            pass
        return None

    async def upload(self, filepath: str, filename: str) -> str | None:
        import cloudinary.uploader
        cloudinary.config(
            cloud_name=self.cloud_name,
            api_key=self.api_key,
            api_secret=self.api_secret,
        )
        try:
            result = cloudinary.uploader.upload(
                filepath,
                public_id=filename,
                folder="AH LCG - ZH",
                unique_filename=False,
            )
            return result.get("secure_url")
        except Exception:
            return None

class ImgBBUploader(ImageUploader):
    def __init__(self, api_key: str):
        self.api_key = api_key

    async def check_exists(self, filename: str) -> str | None:
        return None  # ImgBB 不支持存在性检查

    async def upload(self, filepath: str, filename: str) -> str | None:
        with open(filepath, "rb") as f:
            b64 = base64.b64encode(f.read()).decode()
        async with httpx.AsyncClient() as client:
            resp = await client.post(
                "https://api.imgbb.com/1/upload",
                data={"key": self.api_key, "image": b64, "name": filename},
                timeout=60,
            )
        if resp.status_code == 200:
            return resp.json()["data"]["url"]
        return None

class LocalUploader(ImageUploader):
    def __init__(self, port: int = 8234, host: str = "localhost"):
        self.port = port
        self.host = host

    async def check_exists(self, filename: str) -> str | None:
        return None

    async def upload(self, filepath: str, filename: str) -> str | None:
        return f"http://{self.host}:{self.port}/sheets/{filename}"

def create_uploader(config: dict) -> ImageUploader:
    """工厂函数：根据配置创建上传器"""
    host = config.get("image_host", "local")
    if host == "cloudinary":
        return CloudinaryUploader(
            cloud_name=config["cloud_name"],
            api_key=config["api_key"],
            api_secret=config["api_secret"],
        )
    elif host == "imgbb":
        return ImgBBUploader(api_key=config["imgbb_api_key"])
    else:
        return LocalUploader(port=config.get("localhost_port", 8234))
```

```python
def replace_chinese_card_urls(
    chinese_root: Path,
    url_mapping: dict[str, dict],  # {arkhamdb_id: {card_id, face_url, back_url, deck_key, grid_w, grid_h}}
) -> list[str]:
    """
    替换中文 SCED-downloads 中卡牌的 URL。
    遍历 chinese_root 下所有 JSON，按 GMNotes.id 匹配后更新。
    返回被修改的文件列表。
    """
    modified: list[str] = []
    for json_file in sorted(chinese_root.rglob("*.json")):
        try:
            data = json.loads(json_file.read_text(encoding="utf-8"))
        except (json.JSONDecodeError, UnicodeDecodeError):
            continue

        if data.get("Name") != "Card":
            continue

        gmnotes = data.get("GMNotes", "{}")
        try:
            gm = json.loads(gmnotes)
            card_id = gm.get("id", "")
        except json.JSONDecodeError:
            continue

        if card_id not in url_mapping:
            continue

        mapping = url_mapping[card_id]
        new_deck_key = mapping["deck_key"]
        data["CardID"] = mapping["card_id"]

        # 替换 CustomDeck
        old_deck_key = list(data.get("CustomDeck", {}).keys())[0] if data.get("CustomDeck") else None
        if old_deck_key and old_deck_key in data.get("CustomDeck", {}):
            sheet = data["CustomDeck"].pop(old_deck_key)
            sheet["FaceURL"] = mapping["face_url"]
            sheet["BackURL"] = mapping.get("back_url", sheet.get("BackURL", ""))
            sheet["NumWidth"] = mapping.get("grid_w", sheet.get("NumWidth", 10))
            sheet["NumHeight"] = mapping.get("grid_h", sheet.get("NumHeight", 1))
            data["CustomDeck"][new_deck_key] = sheet
        else:
            data["CustomDeck"] = {
                new_deck_key: {
                    "FaceURL": mapping["face_url"],
                    "BackURL": mapping.get("back_url", ""),
                    "NumWidth": mapping.get("grid_w", 10),
                    "NumHeight": mapping.get("grid_h", 1),
                    "Type": 0,
                    "UniqueBack": mapping.get("unique_back", False),
                    "BackIsHidden": True,
                }
            }

        json_file.write_text(json.dumps(data, ensure_ascii=False, indent=2), encoding="utf-8")
        modified.append(str(json_file.relative_to(chinese_root)))

    return modified
```

```tsx
import { createContext, useContext, useState, useCallback, ReactNode } from 'react';
import { User } from '../types';

interface AuthContextType {
  user: User | null;
  token: string | null;
  login: (username: string, password: string) => Promise<void>;
  logout: () => void;
  isAdmin: boolean;
}

const AuthContext = createContext<AuthContextType>(null!);

export function AuthProvider({ children }: { children: ReactNode }) {
  const [user, setUser] = useState<User | null>(null);
  const [token, setToken] = useState<string | null>(localStorage.getItem('token'));

  const loginFn = useCallback(async (username: string, password: string) => {
    const { login } = await import('../api/auth');
    const data = await login(username, password);
    localStorage.setItem('token', data.token);
    setToken(data.token);
    setUser({ id: data.user_id, username: data.username, role: data.role, is_active: true });
  }, []);

  const logout = useCallback(() => {
    localStorage.removeItem('token');
    setToken(null);
    setUser(null);
  }, []);

  return (
    <AuthContext.Provider value={{
      user, token, login: loginFn, logout,
      isAdmin: user?.role === '管理员',
    }}>
      {children}
    </AuthContext.Provider>
  );
}

export const useAuth = () => useContext(AuthContext);
```

```tsx
import { fetchPendingReviews, batchApprove, batchReject } from '../api/admin';

      <Space>
        <Button type="primary" onClick={handleApprove} disabled={selected.length === 0}>
```

- [ ] **Step 4: 编写 pages/PublishPage.tsx（分步发布界面）**

```tsx
import { useState } from 'react';
import { Card, Steps, Button, message, Upload, Space, Input, Descriptions } from 'antd';
import { UploadOutlined } from '@ant-design/icons';
import { step1GenerateSheets, step2Upload, step5UploadTTSJson, step6ReplaceUrls } from '../api/admin';

export default function PublishPage() {
  const [current, setCurrent] = useState(0);
  const [batchId, setBatchId] = useState('');
  const [sheets, setSheets] = useState<any[]>([]);
  const [urlMapping, setUrlMapping] = useState<Record<string, unknown> | null>(null);
  const [modifiedCount, setModifiedCount] = useState(0);
  const [loading, setLoading] = useState(false);

  const handleStep1 = async () => {
    setLoading(true);
    try {
      const data = await step1GenerateSheets(batchId);
      setSheets(data.generated_sheets);
      message.success(`生成 ${data.total_sheets} 个精灵图，${data.total_cards} 张卡`);
      setCurrent(1);
    } catch (e: any) { message.error(e?.response?.data?.detail || '生成失败'); }
    finally { setLoading(false); }
  };

  const handleStep2 = async () => {
    setLoading(true);
    try {
      const data = await step2Upload(sheets, { image_host: 'local' });
      message.success(`上传完成: ${Object.keys(data.urls).length} 个文件`);
      // 导出 TTS JSON 下载
      setCurrent(2);
    } catch { message.error('上传失败'); }
    finally { setLoading(false); }
  };

  const handleTTSJsonUpload = async (file: File) => {
    const data = await step5UploadTTSJson(file);
    setUrlMapping(data.url_mapping);
    message.success(`提取 ${data.total_cards} 张卡的 Steam URL`);
    setCurrent(4);
    return false; // 阻止默认上传
  };

  const handleStep6 = async () => {
    if (!urlMapping) return;
    setLoading(true);
    try {
      const data = await step6ReplaceUrls(urlMapping);
      setModifiedCount(data.total_modified);
      message.success(`替换完成: ${data.total_modified} 个文件`);
      setCurrent(5);
    } catch { message.error('替换失败'); }
    finally { setLoading(false); }
  };

  const steps = [
    { title: '生成精灵图', content: (
      <Space direction="vertical">
        <Input placeholder="批次ID" value={batchId} onChange={e => setBatchId(e.target.value)} style={{ width: 200 }} />
        <Button type="primary" onClick={handleStep1} loading={loading}>开始生成</Button>
      </Space>
    )},
    { title: '上传图床', content: (
      <Button type="primary" onClick={handleStep2} loading={loading}>上传到图床</Button>
    )},
    { title: '下载TTS存档', content: (
      <Button type="primary" onClick={() => setCurrent(3)}>已下载，下一步</Button>
    )},
    { title: '上传替换后JSON', content: (
      <Upload beforeUpload={handleTTSJsonUpload} maxCount={1} accept=".json">
        <Button icon={<UploadOutlined />}>上传 TTS 导出 JSON</Button>
      </Upload>
    )},
    { title: '替换中文包URL', content: (
      <Button type="primary" onClick={handleStep6} loading={loading} disabled={!urlMapping}>执行替换</Button>
    )},
    { title: '完成', content: (
      <Descriptions column={1}>
        <Descriptions.Item label="修改文件数">{modifiedCount}</Descriptions.Item>
        <Descriptions.Item label="状态">请在 SCED-downloads 目录检查变更并提交 PR</Descriptions.Item>
      </Descriptions>
    )},
  ];

  return (
    <Card title="发布管理">
      <Steps current={current} items={steps.map(s => ({ title: s.title }))} style={{ marginBottom: 32 }} />
      <Card>{steps[current].content}</Card>
    </Card>
  );
}
```
